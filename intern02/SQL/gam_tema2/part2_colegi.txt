--3. Diandra: Scrieti o interogare ce ar afisa  regiunea cu cele mai putine oficii.
SELECT REGIUNE,
       OFICII 
FROM (SELECT REGIUNE,
             COUNT(COD_OFICIU) as OFICII 
      FROM GAM_OFICII 
      GROUP BY REGIUNE 
      ORDER BY COUNT(COD_OFICIU)) 
WHERE ROWNUM =1;
